<script>
// ══════════════════════════════════════════════════
//  ENGINE DE ANÁLISE
// ══════════════════════════════════════════════════

function avg(arr){ return arr.length ? arr.reduce((a,b)=>a+b,0)/arr.length : 0; }
function fmt(n){ return Number(n).toFixed(2); }
function fmtDate(d){ return d ? d.split('-').reverse().join('/') : ''; }

// Poisson base function
function poisson(k, lambda) {
  if (lambda === 0) return k === 0 ? 1 : 0;
  let p = Math.exp(-lambda);
  for (let i = 1; i <= k; i++) p *= lambda / i;
  return p;
}

// Calculate Over lines based on Poisson distribution
function getOvers(lambda, lines) {
  let overs = [];
  lines.forEach(line => {
    let probUnderOrEqual = 0;
    const kMax = Math.floor(line);
    for(let i=0; i<=kMax; i++){
      probUnderOrEqual += poisson(i, lambda);
    }
    const probOver = 1 - probUnderOrEqual;
    overs.push({
      line: 'Over ' + line,
      prob: probOver * 100,
      odd: probOver > 0 ? (1 / probOver) : 0
    });
  });
  return overs;
}

function getStatsTime(time) {
  const jogosValidos = DADOS_2026.jogos.filter(j => j.cantos !== null);
  const casa   = jogosValidos.filter(j => j.mandante  === time);
  const fora   = jogosValidos.filter(j => j.visitante === time);
  const todos  = [...casa,...fora];

  return {
    time,
    jogos: todos.length,
    jogosData: todos,
    media: {
      ht_casa:  avg(casa.map(j => j.cantos.ht.m)),
      ht_casa_c: avg(casa.map(j => j.cantos.ht.v)), // concedido
      ht_fora:  avg(fora.map(j => j.cantos.ht.v)),
      ht_fora_c: avg(fora.map(j => j.cantos.ht.m)), // concedido
      ft_casa:  avg(casa.map(j => j.cantos.ft.m)),
      ft_casa_c: avg(casa.map(j => j.cantos.ft.v)),
      ft_fora:  avg(fora.map(j => j.cantos.ft.v)),
      ft_fora_c: avg(fora.map(j => j.cantos.ft.m)),
      ht_geral: avg(todos.map(j => j.mandante===time ? j.cantos.ht.m : j.cantos.ht.v)),
      ht_geral_c: avg(todos.map(j => j.mandante===time ? j.cantos.ht.v : j.cantos.ht.m)),
      ft_geral: avg(todos.map(j => j.mandante===time ? j.cantos.ft.m : j.cantos.ft.v)),
      ft_geral_c: avg(todos.map(j => j.mandante===time ? j.cantos.ft.v : j.cantos.ft.m))
    }
  };
}

function getRanking() {
  return DADOS_2026.times
    .map(t => getStatsTime(t))
    .filter(t => t.jogos > 0)
    .sort((a,b) => (b.media.ft_geral - b.media.ft_geral_c) - (a.media.ft_geral - a.media.ft_geral_c));
}

function projecaoJogo(mandante, visitante) {
  const m = getStatsTime(mandante);
  const v = getStatsTime(visitante);
  const jogosValidos = DADOS_2026.jogos.filter(j => j.cantos !== null);

  // Médias Gerais do Campeonato
  const l_htHome = avg(jogosValidos.map(j => j.cantos.ht.m));
  const l_htAway = avg(jogosValidos.map(j => j.cantos.ht.v));
  const l_ftHome = avg(jogosValidos.map(j => j.cantos.ft.m));
  const l_ftAway = avg(jogosValidos.map(j => j.cantos.ft.v));

  // -------- CÁLCULO SNIPER (Poisson Bivariado p/ Overs) --------

  // Força Ofensiva (Cantos Gerados)
  const atkHomeHT = m.media.ht_casa / (l_htHome||1);
  const atkAwayHT = v.media.ht_fora / (l_htAway||1);
  const atkHomeFT = m.media.ft_casa / (l_ftHome||1);
  const atkAwayFT = v.media.ft_fora / (l_ftAway||1);

  // Fragilidade Defensiva (Cantos Concedidos)
  const defHomeHT = m.media.ht_casa_c / (l_htAway||1);
  const defAwayHT = v.media.ht_fora_c / (l_htHome||1);
  const defHomeFT = m.media.ft_casa_c / (l_ftAway||1);
  const defAwayFT = v.media.ft_fora_c / (l_ftHome||1);

  // Cantos Esperados (xCorners)
  const expHomeHT = atkHomeHT * defAwayHT * l_htHome;
  const expAwayHT = atkAwayHT * defHomeHT * l_htAway;
  const expTotalHT = expHomeHT + expAwayHT;
  
  const expHomeFT = atkHomeFT * defAwayFT * l_ftHome;
  const expAwayFT = atkAwayFT * defHomeFT * l_ftAway;
  const expTotalFT = expHomeFT + expAwayFT;

  // Calculo simplificado da Confiança baseada na Disparidade vs Variância
  // Diferença entre o modelo avançado Poisson e o somatório base (quanto mais desvia, maior a intervenção)
  const confFT = Math.min(99, Math.max(50, 100 - (Math.abs((m.media.ft_casa+v.media.ft_fora) - expTotalFT)*10)));

  // Mercados "Fair Odds" (Handicaps)
  const oversHT = getOvers(expTotalHT, [3.5, 4.5, 5.5]);
  const oversFT = getOvers(expTotalFT, [8.5, 9.5, 10.5, 11.5]);

  return { expTotalHT, expTotalFT, confFT, oversHT, oversFT, m, v, pow: { atkHomeFT, atkAwayFT, defHomeFT, defAwayFT } };
}

  const ftt_m = parseInt(document.getElementById('inp-ft-m').value);

  if(isNaN(htt_m) || isNaN(htt_v) || isNaN(ftt_m) || isNaN(ftt_v)) {

  const dt = new Date();
  const dataHoje = dt.toISOString().split('T')[0];

    rodada: DADOS_2026.totalRodadas + 1,
    calibrado: true,

  document.getElementById('inp-ft-m').value = '';
  document.getElementById('inp-ft-v').value = '';

function getAppContext() {
  const mode = document.querySelector('.view.active').id;
  let ctx = "Apostador está na seção: " + mode + ".\n";
  if (mode === 'view-projecao') {
    const m = document.getElementById('sel-mandante').value;
    const v = document.getElementById('sel-visitante').value;
    if (m && v && m!==v) {
      const p = projecaoJogo(m, v);
      ctx += `[PROJEÇÃO ATIVA] Partida: ${m} (CASA) vs ${v} (FORA)\n`;
      const jogosLiga = DADOS_2026.jogos.filter(j => j.cantos !== null);
      const mediaLiga = (jogosLiga.reduce((sum, j) => sum + j.cantos.ht.m, 0) / jogosLiga.length).toFixed(2);
      ctx += `- xCorners Real (FT Poisson): ${fmt(p.expTotalFT)} escanteios.\n`;
      ctx += `- Média Liga HT Home: ${mediaLiga}\n`;
      ctx += `- ATK Escanteios ${m}: Força ${fmt(p.pow.atkHomeFT)}x\n`;
      ctx += `- VULNERABILIDADE Defesa ${m}: Fator ${fmt(p.pow.defHomeFT)}x\n`;
      ctx += `- ATK Escanteios ${v}: Força ${fmt(p.pow.atkAwayFT)}x\n`;
      ctx += `- VULNERABILIDADE Defesa ${v}: Fator ${fmt(p.pow.defAwayFT)}x\n`;
      ctx += `Linhas Handicap Fornecidas: ` + p.oversFT.map(o => `${o.line} prob = ${fmt(o.prob)}%`).join(' | ') + `\n`;
    }
  }
  return ctx;
}
